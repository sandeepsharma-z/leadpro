📊 Publish to Google Sheet

Yeh 3-step setup ek baar karni hai:
STEP 1 — Google Sheet banao
sheets.google.com → New sheet banao → naam kuch bhi rakho
STEP 2 — Apps Script deploy karo
Sheet mein: Extensions → Apps Script → yeh code paste karo:
function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var leads = JSON.parse(e.postData.contents).leads;
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(['Name','Phone','Email','Address','City','Country',
      'Category','Rating','Reviews','Lead Type','SEO Score',
      'SEO Issues','Website','Maps Link','Added On']);
  }
  leads.forEach(function(l) {
    sheet.appendRow([l.name,l.phone,l.email,l.address,l.city,l.country,
      l.category,l.rating,l.reviews,l.lead_type,l.seo_score,
      l.seo_issues,l.website_url,l.gmb_url,new Date().toLocaleString()]);
  });
  return ContentService
    .createTextOutput(JSON.stringify({ok:true,added:leads.length}))
    .setMimeType(ContentService.MimeType.JSON);
}
Phir: Deploy → New deployment → Web app → Execute as: Me → Access: Anyone → Deploy → URL copy karo
